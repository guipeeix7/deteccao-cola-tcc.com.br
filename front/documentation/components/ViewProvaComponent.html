import { Component, Input } from '@angular/core';
import { Prova } from '../../prova/shared/prova';
import { ProvaService } from '../shared/prova.service';
import { environment } from 'src/environments/environment';

@Component({
  selector: 'app-view-prova',
  templateUrl: './view-prova.component.html',
  styleUrls: ['./view-prova.component.scss']
})
export class ViewProvaComponent {
  constructor(private provaService: ProvaService){}
  @Input() dataId : number;

  private API_URL = environment.API_URL;

  ProvaData:Prova;
  ngOnInit(){
    this.getData();
  }

  viewQuestao(QuestionId){
    // alert()
    var popupWinWidth = 800
    var popupWinHeight = 600
    var left = (screen.width - popupWinWidth) / 2;
    var top = (screen.height - popupWinHeight) / 2;

    var url = window.location.origin+'/#/questao/update/'+QuestionId;
    console.log(screen.width)
    window.open(url, '_blank', 'location=yes,height='+popupWinHeight+',width='+popupWinWidth+',scrollbars=yes,status=yes,left='+left+',top='+top); // Open new window
    // this.externalWindow = window.open('/questao/update/'+QuestionId, '/questao/update/'+QuestionId, 'width=800,height=600');

    // window.open('https://google.com')
    // window.open(')
  }

  getData(){
    this.provaService.getDataById(this.dataId).subscribe((response) =>{
      this.ProvaData = response['data'];
    })
  }
}
