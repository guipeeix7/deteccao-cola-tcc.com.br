import { Component, forwardRef } from '@angular/core';
import { FormBuilder, FormGroup, NG_VALUE_ACCESSOR, ValidationErrors, Validators } from '@angular/forms';
import { ActivatedRoute } from '@angular/router';
import { BaseService } from 'src/app/shared/base.service';
import Swal from 'sweetalert2';
import { AssuntoService } from '../shared/assunto.service';
import { Assunto } from '../shared/assunto';
import { RolesService } from 'src/app/services/roles-permissions/roles/roles.service';
import { PermissionsService } from 'src/app/services/roles-permissions/permissions/permissions.service';
import { Permissions } from 'src/app/services/roles-permissions/permissions/permissions';
import { NgSelectModule } from '@ng-select/ng-select';
import { Observable, Subject, Subscription, catchError, distinctUntilChanged, of, switchMap, tap } from 'rxjs';
import { List } from 'src/app/shared/models/list';

@Component({
  selector: 'app-update-assunto',
  templateUrl: './update-assunto.component.html',
  styleUrls: ['./update-assunto.component.scss'],

})
export class UpdateAssuntoComponent {
  constructor(private permissionService:PermissionsService, private rolesService: RolesService, private fb: FormBuilder , private baseService: BaseService, private assuntoService : AssuntoService, private route: ActivatedRoute ){}

  form!: FormGroup;
  loading = false;
  disabled = false;
  assunto : Assunto;
  errorMessages: ValidationErrors;
  dataId : number;
  numberOfItemsFromEndBeforeFetchingMore = 5;
  lastCreatedItem: number = 0;
  assuntoList: Permissions[] | undefined;
  permissionsOfRole: Permissions[];
  page:number = 1;
  query:string|'' = '' ;
  totalItemsInDatabase:number = 0;
  selectedItems = []
  peopleInput = new Subject<string>();
  defaultItens = []
  assuntoForm = this.fb.group({
    nomeAssunto: ['',{
      validators: [Validators.compose([Validators.required])]
    }],
    siglaAssunto: [''],
  });

  get nomeAssunto(){ return this.assuntoForm.get('nomeAssunto')}
  get siglaAssunto() { return this.assuntoForm.get('siglaAssunto'); }

  getData(resp){
    this.assunto = resp
    this.nomeAssunto.setValue(this.assunto.nomeAssunto);
    this.siglaAssunto.setValue(this.assunto.siglaAssunto);
  }
  updateSubjetList(lastId){
    this.lastCreatedItem  = lastId
  }

  refreshData(page:number, query:string):Observable<List> {
    return this.baseService.listAll('assunto', page, query, 'nomeAssunto')
  }

  updateSelectData(resp){
    this.assuntoList = resp['data']
    this.totalItemsInDatabase = resp['meta'].total
    this.loading = false;
  }

  ngOnInit(){
    this.dataId = Number(this.route.snapshot.paramMap.get('id'));
    // this.assuntoService.getDataById(this.dataId).subscribe((resp) => {this.getData(resp['data'])})
    this.route.params.subscribe(
      params => {
        const id = this.route.snapshot.paramMap.get('id');
          this.assunto = null;
          this.assuntoService.getDataById(Number(id)).subscribe((resp) => {this.getData(resp['data'])
          this.assunto.id = Number(id)
          this.dataId = Number(id)
        })
      }
      );
  }

  toggleSubmit(){
    if(this.loading == true){
      this.loading = false;
      this.disabled = false;
    }else{
      this.loading = true;
      this.disabled = true;
    }
  }

  onSubmit(){
    this.toggleSubmit()
    this.assuntoService.updateDataInputs(this.dataId, this.assuntoForm.value).subscribe(() => {
      this.assuntoService.getDataById(this.dataId ).subscribe((resp) => {this.getData(resp['data'])})
      this.toggleSubmit()
      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      })
    })
  }
}
