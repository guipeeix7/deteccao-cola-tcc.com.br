import { Component, forwardRef } from '@angular/core';
import { FormBuilder, FormGroup, NG_VALUE_ACCESSOR, ValidationErrors, Validators } from '@angular/forms';
import { ActivatedRoute } from '@angular/router';
import { BaseService } from 'src/app/shared/base.service';
import Swal from 'sweetalert2';
import { AreaService } from '../shared/area.service';
import { Area } from '../shared/area';
import { RolesService } from 'src/app/services/roles-permissions/roles/roles.service';
import { PermissionsService } from 'src/app/services/roles-permissions/permissions/permissions.service';
import { Permissions } from 'src/app/services/roles-permissions/permissions/permissions';
import { NgSelectModule } from '@ng-select/ng-select';
import { Observable, Subject, Subscription, catchError, distinctUntilChanged, of, switchMap, tap } from 'rxjs';
import { List } from 'src/app/shared/models/list';

@Component({
  selector: 'app-update-area',
  templateUrl: './update-area.component.html',
  styleUrls: ['./update-area.component.scss'],

})
export class UpdateAreaComponent {
  constructor(private permissionService:PermissionsService, private rolesService: RolesService, private fb: FormBuilder , private baseService: BaseService, private areaService : AreaService, private route: ActivatedRoute ){}

  form!: FormGroup;
  loading = false;
  disabled = false;
  area : Area;
  errorMessages: ValidationErrors;
  dataId : number;
  numberOfItemsFromEndBeforeFetchingMore = 5;

  areaList: Permissions[] | undefined;
  permissionsOfRole: Permissions[];
  page:number = 1;
  query:string|'' = '' ;
  totalItemsInDatabase:number = 0;
  selectedItems = []
  peopleInput = new Subject<string>();
  defaultItens = [92089939,8]
  areaForm = this.fb.group({
    nomeArea: ['',{
      validators: [Validators.compose([Validators.required])]
    }],
    siglaArea: [''],
  });

  get nomeArea(){ return this.areaForm.get('nomeArea')}
  get siglaArea() { return this.areaForm.get('siglaArea'); }

  getData(resp){
    this.area = resp
    this.nomeArea.setValue(this.area.nomeArea);
    this.siglaArea.setValue(this.area.siglaArea);
  }

  refreshData(page:number, query:string):Observable<List> {
    return this.baseService.listAll('area', page, query, 'nomeArea')
  }

  updateSelectData(resp){
    this.areaList = resp['data']
    this.totalItemsInDatabase = resp['meta'].total
    this.loading = false;
  }

  ngOnInit(){
    this.dataId = Number(this.route.snapshot.paramMap.get('id'));
    this.areaService.getDataById(this.dataId).subscribe((resp) => {this.getData(resp['data'])})
  }

  toggleSubmit(){
    if(this.loading == true){
      this.loading = false;
      this.disabled = false;
    }else{
      this.loading = true;
      this.disabled = true;
    }
  }

  onSubmit(){
    this.toggleSubmit()
    this.areaService.updateDataInputs(this.dataId, this.areaForm.value).subscribe(() => {
      this.areaService.getDataById(this.dataId ).subscribe((resp) => {this.getData(resp['data'])})

      this.toggleSubmit()

      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      })
    })
  }
}
