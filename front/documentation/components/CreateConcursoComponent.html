import { Component } from '@angular/core';
import { FormBuilder, ValidationErrors, Validators } from '@angular/forms';
import { BaseService } from 'src/app/shared/base.service';
import Swal from 'sweetalert2';
import { CustomAsyncValidatorss } from 'src/app/shared/unique-validator.directive';
import { HttpEvent, HttpEventType } from '@angular/common/http';
import { Observable, Subject, last, map } from 'rxjs';
import { List } from 'src/app/services/list/list';
import { convertReactiveIntoformData } from 'src/app/shared/utils'
@Component({
  selector: 'app-create-concurso',
  templateUrl: './create-concurso.component.html',
  styleUrls: ['./create-concurso.component.scss']
})
export class CreateConcursoComponent {
  constructor(
    private fb: FormBuilder,
     private baseService: BaseService,
  ) { }
  loading = false;
  disabled = false;
  errorMessages: ValidationErrors;
  progress:number = 0
  file:File;
  bancaInput = new Subject<string>();
  defaultItens : Array<any>|null;
  numberOfItemsFromEndBeforeFetchingMore = 5;
  totalItemsInDatabase:number = 0;
  selectedItems = []
  bancaList
  preview:any;
  page:number = 1;
  query:string|'' = '' ;

  concursoForm = this.fb.group({
    notaMinima: [null],
    editalUrl: [null],
    realizacaoEm: [null],
    abertoEm: [null],
    updatedAt: [null],
    createdAt: [null],
    siglaConcurso: [null],
    nomeConcurso: ['',{
      validators: [Validators.compose([Validators.required]) ] ,
      asyncValidators: [
        CustomAsyncValidatorss.uniqueNameValidator(this.baseService,  'nomeConcurso', 'concurso-verify-name')
      ],
      updateOn: 'blur'
    }],
    idConcurso: [null],
    escolaridade: [null],
    imagemConcurso: [null],
    numeroVagas: [null],
    possuiTesteAptidaoFisica: [null],
    idBanca: [null]
  });

  get notaMinima() {return this.concursoForm.get('notaMinima');}
  get editalUrl() {return this.concursoForm.get('editalUrl');}
  get realizacaoEm() {return this.concursoForm.get('realizacaoEm');}
  get abertoEm() {return this.concursoForm.get('abertoEm');}
  get updatedAt() {return this.concursoForm.get('updatedAt');}
  get createdAt() {return this.concursoForm.get('createdAt');}
  get siglaConcurso() {return this.concursoForm.get('siglaConcurso');}
  get nomeConcurso() {return this.concursoForm.get('nomeConcurso');}
  get idConcurso() {return this.concursoForm.get('idConcurso');}
  get escolaridade() {return this.concursoForm.get('escolaridade');}
  get imagemConcurso() {return this.concursoForm.get('imagemConcurso');}
  get numeroVagas() {return this.concursoForm.get('numeroVagas');}
  get possuiTesteAptidaoFisica() {return this.concursoForm.get('possuiTesteAptidaoFisica');}
  get idBanca() {return this.concursoForm.get('idBanca');}

  ngOnInit() {
  }


  refreshData(page:number, query:string):Observable<List> {
    return this.baseService.listAll('area', page, query, 'nomeArea')
  }

  toggleSubmit() {
    if (this.loading == true) {
      this.loading = false;
      this.disabled = false;
    } else {
      this.loading = true;
      this.disabled = true;
    }
  }

  get isValid() {
    return this.concursoForm.controls['nomeConcurso'].valid;
  }

  get isTouched() {
    return this.concursoForm.controls['nomeConcurso'].touched;
  }


  private getEventMessage(event: HttpEvent<any>, file: File) {
    if (event.type == HttpEventType.UploadProgress) {
      this.progress = event.total ? Math.round(100 * event.loaded / event.total) : 0;
    }
  }

  onSubmit() {
    this.toggleSubmit()

    let formData = new FormData();
    formData = convertReactiveIntoformData(formData, this.concursoForm);

    this.baseService.createItemWithFile('concurso', formData).pipe(
      map(event => this.getEventMessage(event, this.file)),
      last(),
    ).subscribe(resp => {
      console.log(resp)
      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      }).then(() => {
        // this.router.navigate(['concursos/update/' + resp['data']['id']])
      })
      this.toggleSubmit()
    })
  }

}
