import { Component } from '@angular/core';
import { FormBuilder, ValidationErrors, Validators } from '@angular/forms';
import { BaseService } from 'src/app/shared/base.service';
import Swal from 'sweetalert2';
import { CustomAsyncValidatorss } from 'src/app/shared/unique-validator.directive';
import { HttpEvent, HttpEventType } from '@angular/common/http';
import { Observable, Subject, last, map } from 'rxjs';
import { List } from 'src/app/services/list/list';
import { convertReactiveIntoformData ,convertInitData } from 'src/app/shared/utils'
import { Curso } from '../shared/curso';
import { environment } from 'src/environments/environment';
import { CursoService } from '../shared/curso.service';
import {ActivatedRoute} from '@angular/router'
import { Banca } from '../../banca/shared/banca';
import * as Editor from 'ckeditor/build/ckeditor';

@Component({
  selector: 'app-update-curso',
  templateUrl: './update-curso.component.html',
  styleUrls: ['./update-curso.component.scss']
})
export class UpdateCursoComponent {
  constructor(private fb: FormBuilder , private baseService: BaseService, private cursoService : CursoService, private route: ActivatedRoute ){}
  private API_URL = environment.API_URL;

  loading = false;
  disabled = false;
  errorMessages: ValidationErrors;
  progress:number = 0
  file:File;
  bancaInput = new Subject<string>();
  defaultItens : Array<any>|null;
  numberOfItemsFromEndBeforeFetchingMore = 5;
  totalItemsInDatabase:number = 0;
  selectedItems = []
  bancaList
  preview:any;
  page:number = 1;
  query:string|'' = '' ;
  public Editor = Editor
  dataId:number
  curso:Curso

  cursoForm = this.fb.group({
    resumo: ['', Validators.required],
    descricaoCompleta: ['', Validators.required],
    precoAtual: ['', [Validators.required, Validators.pattern(/^\d+(\.\d{1,2})?$/)]],
    precoPromocao: ['', [Validators.required, Validators.pattern(/^\d+(\.\d{1,2})?$/)]],
    siglaCurso: ['', Validators.required],
    nomeCurso: ['', [Validators.required, Validators.maxLength(180)]],
    imagemCurso: [''],
    idConcurso: ['']
  });


  get resumo() {return this.cursoForm.get('resumo'); }
  get descricaoCompleta() {return this.cursoForm.get('descricaoCompleta');}
  get precoAtual() {return this.cursoForm.get('precoAtual');}
  get precoPromocao() {return this.cursoForm.get('precoPromocao');}
  get siglaCurso() {return this.cursoForm.get('siglaCurso');}
  get nomeCurso() {return this.cursoForm.get('nomeCurso');}
  get imagemCurso() {return this.cursoForm.get('imagemCurso');}
  get idConcurso() {return this.cursoForm.get('idConcurso');}


  getData(resp){
    this.curso = resp
    this.nomeCurso.setValue(this.curso.nomeCurso);
    this.siglaCurso.setValue(this.curso.siglaCurso);
    if(this.curso.imagemCurso != null){
      this.preview = (
          this.curso.imagemCurso.indexOf('http://') >= 0 ||
          this.curso.imagemCurso.indexOf('https://') >= 0
        ) ?
        this.curso.imagemCurso :
        this.API_URL+'storage/'+this.curso.imagemCurso
    }

    convertInitData(resp, this.cursoForm)
    this.baseService.getById('concurso', resp['idConcurso']).subscribe((resp) => {
      this.selectedItems = (resp['data'])
    })
    this.defaultItens = [this.cursoForm.get('idConcurso').value]
  }

  ngOnInit(){
    this.dataId = Number(this.route.snapshot.paramMap.get('id'));
    this.cursoService.getDataById(this.dataId).subscribe((resp) => {this.getData(resp['data'])})

  }

  toggleSubmit() {
    if (this.loading == true) {
      this.loading = false;
      this.disabled = false;
    } else {
      this.loading = true;
      this.disabled = true;
    }
  }

  get isValid() {
    return this.cursoForm.controls['nomeCurso'].valid;
  }

  get isTouched() {
    return this.cursoForm.controls['nomeCurso'].touched;
  }

  private getEventMessage(event: HttpEvent<any>, file: File) {
    if (event.type == HttpEventType.UploadProgress) {
      this.progress = event.total ? Math.round(100 * event.loaded / event.total) : 0;
    }
  }

  onSubmit() {
    this.toggleSubmit()

    let formData = new FormData();
    formData = convertReactiveIntoformData(formData, this.cursoForm);

    this.baseService.updateItemWithFile('curso/'+this.dataId, formData).pipe(
      map(event => this.getEventMessage(event, this.file)),
      last(),
    ).subscribe(resp => {
      console.log(resp)
      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      }).then(() => {
        // this.router.navigate(['cursos/update/' + resp['data']['id']])
      })
      this.toggleSubmit()
    })
  }

}
