import { HttpEvent, HttpEventType } from '@angular/common/http';
import { Component, Renderer2 } from '@angular/core';
import { FormBuilder, Validators } from '@angular/forms';
import { map, last } from 'rxjs';
import { BaseService } from 'src/app/shared/base.service';
import { CustomAsyncValidatorss } from 'src/app/shared/unique-validator.directive';
import { convertReactiveIntoformData } from 'src/app/shared/utils';
import Swal from 'sweetalert2';

@Component({
  selector: 'app-pdf-modal',
  templateUrl: './pdf-modal.component.html',
  styleUrls: ['./pdf-modal.component.scss']
})
export class PdfModalComponent {
  constructor(
    private fb: FormBuilder,
     private baseService: BaseService,
     private renderer: Renderer2
  ) { }
  loading = false;
  disabled = false;
  progress:number = 0
  file:File;
  showPreview = false
  display = 'block'

  pdfForm = this.fb.group({
    nomePdf: ['',{
      validators: [Validators.compose([Validators.required]) ] ,
      asyncValidators: [
        CustomAsyncValidatorss.uniqueNameValidator(this.baseService,  'nomePdf', 'pdf-verify-name')
      ],
      updateOn: 'blur'
    }],
    caminho: [null, Validators.required],
  });

  get nomePdf() {return this.pdfForm.get('nomePdf');}
  get caminho() {return this.pdfForm.get('caminho');}

  ngOnInit() {
  }

  showPreviewFunction(){
    this.display = 'block'
    this.showPreview = true
    if(this.showPreview){
      this.renderer.selectRootElement('#buttonPdf').click();
    }
  }

  toggleSubmit() {
    if (this.loading == true) {
      this.loading = false;
      this.disabled = false;
    } else {
      this.loading = true;
      this.disabled = true;
    }
  }

  private getEventMessage(event: HttpEvent<any>, file: File) {
    if (event.type == HttpEventType.UploadProgress) {
      this.progress = event.total ? Math.round(100 * event.loaded / event.total) : 0;
    }
  }

  onSubmit() {
    this.toggleSubmit()

    let formData = new FormData();
    formData = convertReactiveIntoformData(formData, this.pdfForm);

    this.baseService.createItemWithFile('pdf', formData).pipe(
      map(event => this.getEventMessage(event, this.file)),
      last(),
    ).subscribe(resp => {
      console.log(resp)
      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      }).then(() => {
        // this.router.navigate(['pdfs/update/' + resp['data']['id']])
        this.nomePdf.setValue('')
        this.caminho.setValue(null)
        this.display = 'none'
      })
      this.toggleSubmit()
    })
  }
}
