import { Component} from '@angular/core';
import { FormGroup, ValidationErrors } from '@angular/forms';
import Swal from 'sweetalert2';
import { ActivatedRoute, Router } from '@angular/router';
import { Location } from '@angular/common';
import { animate, style, transition, trigger } from '@angular/animations';
import { FormBuilder, Validators } from '@angular/forms';
import { RolesService } from 'src/app/services/roles-permissions/roles/roles.service';
import { PermissionsService } from 'src/app/services/roles-permissions/permissions/permissions.service';
import { Observable } from 'rxjs';
import { Permissions } from 'src/app/services/roles-permissions/permissions/permissions';
import { Roles } from 'src/app/services/roles-permissions/roles/roles';
// import { UniqueValidatorDirective, uniqueName } from '../../../../shared/unique-validator.directive';
// import { CreateService } from 'src/app/services/partials/create.service';

@Component({
  selector: 'app-create-permisssion',
  templateUrl: './create-permisssion.component.html',
  styleUrls: ['./create-permisssion.component.scss']
})
export class CreatePermisssionComponent {
  constructor(private fb: FormBuilder, private router: Router, private rolesService: RolesService, private route: ActivatedRoute, private permissionService:PermissionsService,
    ) { }

  loading = false;
  disabled = false;
  rolesList: Roles[] | undefined ;
  rolesOfPermissions: Roles[];
  role: Roles;
  errorMessages: ValidationErrors;

  form = this.fb.group({
    name: ['', { validators: Validators.compose([Validators.required]), updateOn: 'blur', }],
    permissions: ['', ],
  });

  get name() { return this.form.get('name'); }
  get permissions() { return this.form.get('permissions'); }

  ngOnInit(){
    this.rolesService.getAll().subscribe((resp)=>{this.rolesList = resp});
  }

  toggleSubmit(){
    if(this.loading == true){
      this.loading = false;
      this.disabled = false;
    }else{
      this.loading = true;
      this.disabled = true;
    }
  }

  onSubmit(){
    this.toggleSubmit()

    // this.createService.sendData('permission', this.form.value).subscribe(resp =>{
    //   // this.sendData(resp)
    //   }
    // )


    this.toggleSubmit()
      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      }).then(()=>{
        this.router.navigate(['/roles-permissions'])
      })
      this.toggleSubmit()
  }
}
