import { Component } from '@angular/core';
import { FormBuilder, Validators } from '@angular/forms';
import { BaseService } from 'src/app/shared/base.service';
import Swal from 'sweetalert2';
import { convertInitData } from 'src/app/shared/utils'
import { Questao } from '../shared/questao';
import { environment } from 'src/environments/environment';
import { QuestaoService } from '../shared/questao.service';
import {ActivatedRoute} from '@angular/router'
import { Area } from '../../../pages/area/shared/area';
import * as Editor from 'ckeditor/build/ckeditor';
import { Disciplina } from '../../disciplina/shared/disciplina';
import { Posicao } from '../../posicao/shared/posicao';
import { Orgao } from '../../orgaos/shared/orgao';
import { Concurso } from '../../concurso/shared/concurso';
import { Assunto } from '../../assunto/shared/assunto';

@Component({
  selector: 'app-update-questao',
  templateUrl: './update-questao.component.html',
  styleUrls: ['./update-questao.component.scss']
})
export class UpdateQuestaoComponent {
  constructor(private fb: FormBuilder , private baseService: BaseService, private questaoService : QuestaoService, private route: ActivatedRoute ){}
  private API_URL = environment.API_URL;

  loading = false;
  disabled = false;
  file:File;
  defaultItens : Array<any>|null;
  selectedItems = []
  bancaList
  public Editor = Editor
  public Editor2 = Editor
  dataId:number
  questao:Questao
  area:Area
  areaList: Area[];
  disciplinaList:Disciplina[]
  posicaoList:Posicao[]
  orgaoList:Orgao[]
  concursoList:Concurso[]
  assuntoList:Assunto[]

  questaoForm = this.fb.group({
    nomeQuestao: ['', Validators.required],
    descricaoQuestao: ['', Validators.required],
    opcaoCorreta: ['', Validators.required],
    idBanca: ['', Validators.required],
    dataHora: ['', Validators.required],
    resposta: ['', Validators.required],
    idArea: [null,],
    idDisciplina: [null,],
    idPosicao: [null,],
    idOrgao: [null,],
    idConcurso: [null,],
    idAssunto: [null,],
  });

  get nomeQuestao() {return this.questaoForm.get('nomeQuestao'); }
  get descricaoQuestao() {return this.questaoForm.get('descricaoQuestao');}
  get opcaoCorreta() {return this.questaoForm.get('opcaoCorreta');}
  get idBanca() {return this.questaoForm.get('idBanca');}
  get dataHora() {return this.questaoForm.get('dataHora');}
  get resposta() {return this.questaoForm.get('resposta');}
  get idArea() {return this.questaoForm.get('idArea');}
  get idDisciplina() {return this.questaoForm.get('idDisciplina');}
  get idPosicao() {return this.questaoForm.get('idPosicao');}
  get idOrgao() {return this.questaoForm.get('idOrgao');}
  get idConcurso() {return this.questaoForm.get('idConcurso');}
  get idAssunto() {return this.questaoForm.get('idAssunto');}

  getData(resp){
    this.questao = resp

    this.areaList = resp['areas']
    this.disciplinaList = resp['disciplinas']
    this.posicaoList = resp['posicoes']
    this.orgaoList = resp['orgaos']
    this.concursoList = resp['concursos']
    this.assuntoList = resp['assuntos']

    console.log(this.areaList)
    convertInitData(resp, this.questaoForm)

    this.questaoForm.patchValue({idArea: this.areaList.map((x) => {return x['id']})});
    this.questaoForm.patchValue({idDisciplina: this.disciplinaList.map((x) => {return x['id']})});
    this.questaoForm.patchValue({idPosicao: this.posicaoList.map((x) => {return x['id']})});
    this.questaoForm.patchValue({idOrgao: this.orgaoList.map((x) => {return x['id']})});
    this.questaoForm.patchValue({idConcurso: this.concursoList.map((x) => {return x['id']})});
    this.questaoForm.patchValue({idAssunto: this.assuntoList.map((x) => {return x['id']})});



    this.baseService.getById('banca', resp['idBanca']).subscribe((resp) => {
      // this.selectedItems = (resp['data'])
    })



    this.defaultItens = [this.questaoForm.get('idBanca').value]
  }

  ngOnInit(){
    this.dataId = Number(this.route.snapshot.paramMap.get('id'));
    this.questaoService.getDataById(this.dataId).subscribe((resp) => {this.getData(resp['data'])})

  }

  toggleSubmit() {
    if (this.loading == true) {
      this.loading = false;
      this.disabled = false;
    } else {
      this.loading = true;
      this.disabled = true;
    }
  }

  get isValid() {
    return this.questaoForm.controls['nomeQuestao'].valid;
  }

  get isTouched() {
    return this.questaoForm.controls['nomeQuestao'].touched;
  }

  onSubmit() {
    this.toggleSubmit()
    this.questaoService.updateDataInputs(this.dataId, this.questaoForm.value).subscribe(resp => {
      console.log(resp)
      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      }).then(() => {
        // this.router.navigate(['questaos/update/' + resp['data']['id']])
      })
      this.toggleSubmit()
    })
  }

}
