import { Component, Input } from '@angular/core';
import { FormBuilder, FormGroup, ValidationErrors, Validators } from '@angular/forms';
import { ActivatedRoute } from '@angular/router';
import * as Editor from 'ckeditor/build/ckeditor';
import { Subject, Observable } from 'rxjs';
import { List } from 'src/app/services/list/list';
import { PermissionsService } from 'src/app/services/roles-permissions/permissions/permissions.service';
import { RolesService } from 'src/app/services/roles-permissions/roles/roles.service';
import { BaseService } from 'src/app/shared/base.service';
import Swal from 'sweetalert2';
import { Correcao } from '../shared/correcao';
import { CorrecaoService } from '../shared/correcao.service';
import { QuestaoService } from '../../questao/shared/questao.service';
import { Questao } from '../../questao/shared/questao';

@Component({
  selector: 'app-create-question-link',
  templateUrl: './create-question-link.component.html',
  styleUrls: ['./create-question-link.component.scss']
})
export class CreateQuestionLinkComponent {
  constructor(private permissionService:PermissionsService, private rolesService: RolesService, private fb: FormBuilder , private baseService: BaseService, private correcaoService : CorrecaoService,private questionService:QuestaoService , private route: ActivatedRoute ){}
  @Input() cursoId:number;
  form!: FormGroup;
  loading = false;
  disabled = false;
	public videoCollapse = true;
	public pdfCollapse = true;
  correcao : Correcao;
  errorMessages: ValidationErrors;
  dataId : number;
  questao: Questao;
  numberOfItemsFromEndBeforeFetchingMore = 5;
  public Editor = Editor

  page:number = 1;
  query:string|'' = '' ;
  totalItemsInDatabase:number = 0;
  selectedItems = []
  peopleInput = new Subject<string>();
  correcaoForms = this.fb.group({
    textoCorrecao: ['', Validators.required],
    idVideo: [,],
    idPdf: [,],
    idQuestao : ['', Validators.required]
  });

  get textoCorrecao() { return this.correcaoForms.get('textoCorrecao'); }
  get idVideo() { return this.correcaoForms.get('idVideo'); }
  get idPdf() { return this.correcaoForms.get('idPdf'); }
  get idQuestao() { return this.correcaoForms.get('idQuestao'); }

  getData(resp){
    this.questao = resp
    this.idQuestao.setValue(this.questao.id.toString());
  }

  refreshData(page:number, query:string):Observable<List> {
    return this.baseService.listAll('correcao-all', page, query, 'nomeCorrecao')
  }


  ngOnInit(){
    this.dataId = Number(this.route.snapshot.paramMap.get('id'));
    this.questionService.getDataById(this.dataId).subscribe((resp) => {this.getData(resp['data'])})
  }

  toggleSubmit(){
    if(this.loading == true){
      this.loading = false;
      this.disabled = false;
    }else{
      this.loading = true;
      this.disabled = true;
    }
  }
  toggleCollapseds($event){
    // this.videoCollapse = ;

    // this.pdfCollapse = this.;
  }

  handlePdfSelections($event){
    this.idPdf.patchValue($event)
    console.log($event)
  }
  handleVideoSelections($event){
    this.idVideo.patchValue($event)
    console.log($event)
  }

  onSubmit(){
    this.toggleSubmit()
    this.baseService.createItem('correcao', this.correcaoForms.value).subscribe(() => {
      // this.correcaoService.getDataById(this.dataId ).subscribe((resp) => {this.getData(resp['data'])})

      this.toggleSubmit()

      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      })
    })
  }
}
