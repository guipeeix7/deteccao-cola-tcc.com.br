import { Component, EventEmitter , Input, Renderer2, ElementRef} from '@angular/core';
import { BaseItemComponent } from 'src/app/shared/base-item/base-item.component';
import { BaseItem } from 'src/app/shared/base-item/base-item.interface';
import { SelectableItemComponent } from 'src/app/shared/custom-selectable-list/selectable-item/selectable-item.component';
import { Questao } from '../questao';

@Component({
  selector: 'app-questao-item',
  templateUrl: './questao-item.component.html',
  styleUrls: ['./questao-item.component.scss']
})
export class QuestaoItemComponent extends BaseItemComponent{
  constructor(private renderer: Renderer2){
    super();
  }
  @Input() ItemData:Questao;
  isCollapsed = true
  showPreview = false

  showPreviewFunction(){
    this.showPreview = true
    if(this.showPreview){
      this.renderer.selectRootElement('#buttonOpenModal'+this.ItemData.id).click();
    }
  }


}

<ng-container class="text-success">
  <th scope="row">
    <button

      [class.itemSelected]="clicked == true"

      class="btn btn-sm btn-outline-success"
    >
      <span class="fas" [class]="clicked ? 'fa-minus' : 'fa-plus'"></span>
    </button>

    <button CustomModalButton   (click)="showPreviewFunction()" class="btn btn-sm btn-outline-info">
      <span class="fas fa-eye" >
      </span>
    </button>

    <ng-container >
      <app-modal [title]="ItemData.nomeQuestao" style="word-break: break-word;">
        <button id="buttonOpenModal{{ ItemData.id }}" style="visibility:hidden" CustomModalButton ></button>
        <ng-container *ngIf="showPreview">
          <app-view-questao [dataId]="ItemData.id"></app-view-questao>

      </app-modal>

</ng-container>
