import { Component, Input } from '@angular/core';
import { FormBuilder, ValidationErrors, Validators } from '@angular/forms';
import { Router, ActivatedRoute } from '@angular/router';
import { PermissionsService } from 'src/app/services/roles-permissions/permissions/permissions.service';
import { BaseService } from 'src/app/shared/base.service';
import { CustomAsyncValidatorss } from 'src/app/shared/unique-validator.directive';
import Swal from 'sweetalert2';
import * as Editor from 'ckeditor/build/ckeditor';

@Component({
  selector: 'app-create-prova',
  templateUrl: './create-prova.component.html',
  styleUrls: ['./create-prova.component.scss']
})
export class CreateProvaComponent {
  constructor(private fb: FormBuilder, private router: Router, private baseService: BaseService ,
    // private uniqueValidatorDirective: CustomAsyncValidatorss
  ) { }
  defaultItens : Array<any>|null;
  public Editor = Editor

  loading = false;
  disabled = false;
  errorMessages: ValidationErrors;
  provaForms = this.fb.group({
    nomeProva: ['', {
      validators: [Validators.required],
      asyncValidators: [
        CustomAsyncValidatorss.uniqueNameValidator(this.baseService,  'nomeProva', 'prova-verify-name')
      ],
      updateOn: 'blur'

    }],
    descricaoProva: ['', {
      validators: [Validators.required],
    }],
    dataAplicacao: ['', {
      validators: [Validators.required],
    }]
  });

  get nomeProva() { return this.provaForms.get('nomeProva'); }
  get descricaoProva() { return this.provaForms.get('descricaoProva'); }
  get dataAplicacao() { return this.provaForms.get('dataAplicacao'); }

  ngOnInit(){
  }

  toggleSubmit(){
    if(this.loading == true){
      this.loading = false;
      this.disabled = false;
    }else{
      this.loading = true;
      this.disabled = true;
    }
  }

  get isValid(){return this.provaForms.controls['nomeProva'].valid;}

  get isTouched(){return this.provaForms.controls['nomeProva'].touched;}

  onSubmit(){

      this.toggleSubmit()
      this.baseService.createItem('prova', this.provaForms.value).subscribe(resp =>{

      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      }).then(()=>{
        this.router.navigate(['prova/update/'+resp['data']['id']])

      })
      this.toggleSubmit()
    })
  }
}
