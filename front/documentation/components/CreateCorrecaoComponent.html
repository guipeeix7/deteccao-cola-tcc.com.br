import { Component, Input } from '@angular/core';
import { FormBuilder, ValidationErrors, Validators } from '@angular/forms';
import { Router, ActivatedRoute } from '@angular/router';
import { PermissionsService } from 'src/app/services/roles-permissions/permissions/permissions.service';
import { BaseService } from 'src/app/shared/base.service';
import { CustomAsyncValidatorss } from 'src/app/shared/unique-validator.directive';
import Swal from 'sweetalert2';
import * as Editor from 'ckeditor/build/ckeditor';

@Component({
  selector: 'app-create-correcao',
  templateUrl: './create-correcao.component.html',
  styleUrls: ['./create-correcao.component.scss']
})
export class CreateCorrecaoComponent {
  public Editor = Editor

  constructor(private fb: FormBuilder, private router: Router, private baseService: BaseService ,
    // private uniqueValidatorDirective: CustomAsyncValidatorss
  ) { }
  defaultItens : Array<any>|null;
  @Input() idCurso:number|null;

  loading = false;
  disabled = false;
  errorMessages: ValidationErrors;
  correcaoForms = this.fb.group({
    textoCorrecao: [null, Validators.required],
  });

  get textoCorrecao() { return this.correcaoForms.get('textoCorrecao'); }

  ngOnInit(){
    this.correcaoForms.controls['idCurso'].setValue(this.idCurso);
  }

  toggleSubmit(){
    if(this.loading == true){
      this.loading = false;
      this.disabled = false;
    }else{
      this.loading = true;
      this.disabled = true;
    }
  }

  get isValid(){
    return this.correcaoForms.controls['nomeCorrecao'].valid;
  }

  get isTouched(){
    return this.correcaoForms.controls['nomeCorrecao'].touched;
  }

  onSubmit(){
    this.toggleSubmit()
    this.baseService.createItem('correcao', this.correcaoForms.value).subscribe(resp =>{
      Swal.fire({
        position: 'top-end',
        icon: 'success',
        title: 'Registro atualizado com sucesso',
        showConfirmButton: false,
        timer: 2000
      }).then(()=>{
        this.router.navigate(['correcaos/update/'+resp.idCorrecao])

      })
      this.toggleSubmit()
    })
  }
}
