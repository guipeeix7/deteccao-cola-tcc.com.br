import { Component, Output, EventEmitter, Renderer2 } from '@angular/core';
import { FormBuilder } from '@angular/forms';
import { Meta } from '@angular/platform-browser';
import { List } from 'src/app/services/list/list';
import { BaseListComponent } from 'src/app/shared/base-selectable-list/base-selectable-list.component';
import { BaseService } from 'src/app/shared/base.service';
import { Area } from '../../area/shared/area';
import { Banca } from '../../banca/shared/banca';
import { Concurso } from '../../concurso/shared/concurso';
import { Disciplina } from '../../disciplina/shared/disciplina';
import { Orgao } from '../../orgaos/shared/orgao';
import { Posicao } from '../../posicao/shared/posicao';
import { QuestaoService } from '../shared/questao.service';

@Component({
  selector: 'app-selectable-questao-list',
  templateUrl: './selectable-questao-list.component.html',
  styleUrls: ['./selectable-questao-list.component.scss']
})

export class SelectableQuestaoListComponent  extends BaseListComponent{
  constructor(public baseService: BaseService, private fb: FormBuilder, private questaoService : QuestaoService, private renderer: Renderer2,){
    super(baseService);
  }
  @Output() selectedQuestionsEvent = new EventEmitter<number[]>;

  itemName = 'nomeQuestao'
  namedRoute = 'questao'
  isCollapsed=true;
  questaosData : List;
  questaosMeta : Meta;
  page:number = 1;
  query:string = '';
  bancaList : Banca[] = [];
  areaList: Area[] = [];
  disciplinaList:Disciplina[] = [];
  posicaoList:Posicao[] = [];
  orgaoList:Orgao[] = [];
  concursoList:Concurso[] = [];
  showPreview:boolean = false;
  questaoForm = this.fb.group({
    idArea: [null,],
    idBanca: [null,],
    idDisciplina: [null,],
    idPosicao: [null,],
    idOrgao: [null,],
    idConcurso: [null,],
    page : 0,
    per_page : 15,
    nomeQuestao: ['']
  });

  get idArea() {return this.questaoForm.get('idArea');}
  get idBanca() {return this.questaoForm.get('idBanca');}
  get idDisciplina() {return this.questaoForm.get('idDisciplina');}
  get idPosicao() {return this.questaoForm.get('idPosicao');}
  get idOrgao() {return this.questaoForm.get('idOrgao');}
  get idConcurso() {return this.questaoForm.get('idConcurso');}
  get nomeQuestao() {return this.questaoForm.get('nomeQuestao');}
  get pageForms() {return this.questaoForm.get('page');}
  get per_page() {return this.questaoForm.get('per_page');}

  getData(page:number , query:string){
    this.nomeQuestao.setValue(this.query)
    this.pageForms.setValue(this.page)
    this.per_page.setValue(15)

    this.questaoService.listAll(this.namedRoute, page, query, this.itemName, this.questaoForm.value).subscribe(
      (resp) => {
        this.ListData = resp
    })
  }
}
