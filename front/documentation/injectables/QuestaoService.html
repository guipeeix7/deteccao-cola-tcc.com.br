import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { Questao } from './questao';
import { environment } from 'src/environments/environment'
import { List } from 'src/app/services/list/list';


@Injectable({
  providedIn: 'root'
})
export class QuestaoService {
  constructor(private http:HttpClient) { }
  private API_URL = environment.API_URL;

  listAllQuestaos() {
    return this.http.get<any>(this.API_URL+"/questao");
  }

  getDataById(id :number): Observable<any>{
    return this.http.get<Questao>(this.API_URL+'questao'+'/'+id);
  }

  updateDataInputs(id:number, formData:any): Observable<any> {
    return this.http.put<Questao>(this.API_URL+'questao/'+id, formData);
  }
  sendAnswerToServer(selectedPosition:number, selectedId:number, questionId:number){
    return this.http.post<any>(this.API_URL+'questao/user-answer/'+questionId, {'positionSelected' :selectedPosition , 'selectedId':selectedId })
  }

  listAll(urlFragment ,page, query, searchable, forms): Observable<List>  {
    let queryParams = new HttpParams();
    queryParams = queryParams.append("nomeQuestao", forms.nomeQuestao);
    queryParams = queryParams.append("idArea[]",forms.idArea == null ? '' : forms.idArea);
    queryParams = queryParams.append("idBanca[]",forms.idBanca == null ? '' : forms.idBanca);
    queryParams = queryParams.append("idPosicao[]",forms.idPosicao == null ? '' : forms.idPosicao);
    queryParams = queryParams.append("idOrgao[]",forms.idOrgao == null ? '' : forms.idOrgao);
    queryParams = queryParams.append("idConcurso[]",forms.idConcurso == null ? '' : forms.idConcurso);
    queryParams = queryParams.append("idAssunto[]",forms.idAssunto == null ? '' : forms.idAssunto);
    queryParams = queryParams.append("bancaList[]",forms.idConcurso == null ? '' : forms.idConcurso);
    queryParams = queryParams.append("page",forms.page);
    queryParams = queryParams.append("per_page",forms.per_page);
    return this.http.get<List>(this.API_URL+urlFragment, { params:queryParams } );
  }

  getQuestionListByUserResponse(questionList: Questao[]): Observable<List>{
    return this.http.get<List>(this.API_URL+'questao/user/questao-review' );
  }

}
